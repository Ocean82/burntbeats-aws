# CPU-only stem splitting (no CUDA). Reproducible installs: use stem_service/requirements.lock.txt with
#   pip install -r stem_service/requirements.lock.txt --extra-index-url https://download.pytorch.org/whl/cpu
# (see stem_service/Dockerfile and Makefile). For ad-hoc CPU: pip install torch --index-url https://download.pytorch.org/whl/cpu then this file.
#
# IMPORTANT: Always activate the project venv before running pip or python:
#   source .venv/bin/activate   (WSL/Ubuntu/Linux)
#   .venv\Scripts\activate      (Windows)
# Running against a system/user Python that has torch>=2.6 installed will break
# 4-stem fast splits even though demucs_entry.py patches torch.load, because the
# patch runs in the subprocess Python — if that Python is not the venv Python,
# the patch is applied to the wrong torch installation.
#
# Pin torch/torchaudio <2.6: PyTorch 2.6+ defaults torch.load(weights_only=True). Demucs loads
# htdemucs checkpoints with pickle (HTDemucs class); unpickling fails unless weights_only=False,
# and the stock `python -m demucs` CLI does not pass that flag — 2-stem/4-stem Demucs breaks.
# demucs_entry.py patches torch.load to inject weights_only=False + safe_globals allowlist,
# but this only works when the subprocess uses the venv Python (sys.executable).
# Also keep <2.9 so load/save paths avoid torchcodec issues on some hosts (WSL/CI).
torch>=2.0.0,<2.6
torchaudio>=2.0.0,<2.6
soundfile>=0.12.0
# Bundled ffmpeg when system ffmpeg is not on PATH (matrix clip extraction, optional audio-separator env)
imageio-ffmpeg>=0.4.0
demucs>=4.0.0
# Demucs uses DiffQ for some quantized UVR / bag checkpoints
diffq>=0.2.0
# onnxruntime (not onnxruntime-gpu) for CPU-optimized MDX inference; SessionOptions use ORT_ENABLE_ALL + configurable threads
onnxruntime>=1.14.0
# Optional: Intel OpenVINO execution provider for ONNX (set USE_ONNX_OPENVINO=1 in .env). Use a build that ships the EP, e.g. onnxruntime-openvino.
# NumPy 2.x can break older torch wheels (e.g. 2.2); keep <2.6 for scipy compatibility.
numpy>=1.24.0,<2.6
# server_export (fftconvolve, resample, lfilter)
scipy>=1.9.0
fastapi>=0.100.0
uvicorn[standard]>=0.22.0
python-multipart>=0.0.6
# VAD pre-trim uses ONNX (models/silero_vad.onnx) via onnxruntime + PyAV; no silero-vad pip needed
av>=16.0.0

# Optional: upload stems to S3 after each job (S3_ENABLED=true; see docs/ARCHITECTURE-FLOW.md)
boto3>=1.28.0
